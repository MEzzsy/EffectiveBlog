1. 排序

1.1. 快速排序

private void quickSort(int[] A) {
    quickSort(A, 0, A.length - 1);
}

private void quickSort(int[] A, int p, int r) {
    if (p < r) {
        int q = partition(A, p, r);
        quickSort(A, p, q - 1);
        quickSort(A, q + 1, r);
    }
}

private int partition(int[] A, int p, int r) {
    int x = A[r];
    int q = p - 1;
    for (int i = p; i <= r; i++) {
        if (A[i] <= x) {
            q++;
            int t = A[q];
            A[q] = A[i];
            A[i] = t;
        }
    }
    return q;
}

时间复杂度O(n*n)，平均时间复杂度O(nlogn)，空间复杂度O(1)。

1.2. 归并排序

public void mergeSort(int[] A) {
    mergeSort(A, new int[A.length], 0, A.length - 1);
}

private void mergeSort(int[] A, int[] B, int p, int q) {
    if (p >= q)
        return;
    int r = (p + q) / 2;

    mergeSort(A, B, p, r);
    mergeSort(A, B, r + 1, q);

    System.arraycopy(A, p, B, p, q - p + 1);

    int i = r;
    int j = q;
    int k;
    for (k = q; k >= p && i >= p && j >= r + 1; k--) {
        if (B[i] > B[j]) {
            A[k] = B[i];
            i--;
        } else {
            A[k] = B[j];
            j--;
        }
    }
    while (i >= p) A[k--] = B[i--];
    while (j >= r + 1) A[k--] = B[j--];
}

时间复杂度O(nlogn)，空间复杂度O(n)。

1.3. 计数排序

//计数排序
private int[] countingSort(int[] A) {
    int max = Integer.MIN_VALUE;
    for (int i : A) {
        if (i > max)
            max = i;
    }
    int[] B = new int[max + 1];
    int[] C = new int[A.length];
    for (int i : A) B[i]++;
    for (int i = 1; i <= max; i++) B[i] = B[i] + B[i - 1];//A的每个数在C中的位置
    for (int a : A) {
        int p = B[a];
        C[p - 1] = a;
        B[a] = --p;
    }
    return C;
}

时间复杂度O(k+n)，空间复杂度O(k)，k为A中最大数。

1.4. 插入排序

//插入排序
private void insertSort(int[] nums) {
    if (nums == null || nums.length <= 1)
        return;
    for (int i = 1; i < nums.length; i++) {
        int j = i - 1;
        int x = nums[i];
        while (j >= 0 && nums[j] > x) {
            nums[j + 1] = nums[j];
            j--;
        }
        nums[j + 1] = x;
    }
}

时间复杂度O(n*n)，空间复杂度O(1)。

1.5. 堆排序

/**
 * 堆排序
 *
 * @param A
 */
public void heapSort(int[] A) {
    buildHeap(A);
    for (int i = A.length - 1; i >= 0; i--) {
        int t = A[0];
        A[0] = A[i];
        A[i] = t;

        maxHeapify(A, 0, i - 1);
    }
}

/**
 * 建堆
 *
 * @param A
 */
private void buildHeap(int[] A) {
    for (int i = A.length / 2; i >= 0; i--) {
        maxHeapify(A, i, A.length - 1);
    }
}

/**
 * 维护最大堆的性质
 *
 * @param A 排序的数组
 * @param p 起始坐标
 * @param q 结束坐标
 */
private void maxHeapify(int[] A, int p, int q) {
    if (p > q)
        return;
    int l = 2 * p + 1;
    int r = 2 * p + 2;
    int largest;
    if (l <= q && A[l] > A[p]) largest = l;
    else largest = p;
    if (r <= q && A[r] > A[largest]) largest = r;
    if (largest != p) {
        int t = A[p];
        A[p] = A[largest];
        A[largest] = t;
        maxHeapify(A, largest, q);
    }
}

时间复杂度O(nlogn)，空间复杂度O(1)。

2. 二分搜索

/**
 * 二分查找，如果有重复的，返回第一次出现的坐标，如果没有这个数，返回-1。
 *
 * @param A
 * @param num
 * @return
 */
public int binarySearch4(int[] A, int num) {
    if (A == null || A.length == 0)
        return -1;
    int l = 0;
    int r = A.length;
    int mid;
    while (l < r) {
        mid = (l + r) / 2;
        if (A[mid] >= num)
            r = mid;
        else
            l = mid + 1;
    }
    if (l >= A.length)
        return -1;
    return A[l] == num ? l : -1;
}

/**
 * 二分查找，如果有，返回坐标；如果没有，返回合适的插入坐标。非递归。
 *
 * @param A
 * @param num
 * @return
 */
public int binarySearch3(int[] A, int num) {
    if (A == null || A.length == 0)
        return -1;
    int p = 0;
    int q = A.length - 1;
    int mid = -1;
    while (p <= q) {
        mid = (p + q) / 2;
        if (A[mid] == num)
            return mid;
        else if (A[mid] < num)
            p = mid + 1;
        else
            q = mid - 1;
    }
    return mid;
}

/**
 * 二分查找，非递归
 *
 * @param A
 * @param num
 * @return
 */
public int binarySearch2(int[] A, int num) {
    if (A == null || A.length == 0)
        return -1;
    int p = 0;
    int q = A.length - 1;
    int mid;
    while (p <= q) {
        mid = (p + q) / 2;
        if (A[mid] == num)
            return mid;
        else if (A[mid] < num)
            p = mid + 1;
        else
            q = mid - 1;
    }
    return -1;
}

/**
 * 二分查找，递归
 *
 * @param A
 * @param num
 * @return
 */
public int binarySearch(int[] A, int num) {
    if (A == null || A.length == 0)
        return -1;
    return binarySearch(A, num, 0, A.length - 1);
}

private int binarySearch(int[] A, int num, int p, int q) {
    if (p <= q) {
        int mid = (p + q) / 2;
        if (A[mid] == num)
            return mid;
        else if (A[mid] < num)
            return binarySearch(A, num, mid + 1, q);
        else
            return binarySearch(A, num, p, mid - 1);
    }
    return -1;
}

这里二分搜索，如果r=len，那么区间被分为左闭右开[l,mid-1)，[mid+1,r)，所以循环退出条件位l<r；如果len-1，那么区间被分为左闭右闭[l,mid-1]，[mid+1,r]，所以循环退出条件位l<=r。
